#include <stdio.h>
#include <ctype.h>
#include <string.h>

#define SIZE 50

char stack[SIZE];
int top = -1;

void push(char elem);
char pop();
int pr(char symbol);
void infixToPostfix(char infix[], char postfix[]);

void push(char elem) {
    if (top >= SIZE - 1) {
        printf("\nStack Overflow!");
        return;
    }
    stack[++top] = elem;
}

char pop() {
    if (top == -1) {
        return '#';
    }
    return stack[top--];
}

int pr(char symbol) {
    if (symbol == '(') {
        return 0; 
    } else if (symbol == '+' || symbol == '-') {
        return 1;
    } else if (symbol == '*' || symbol == '/') {
        return 2;
    }
    return 0; 
}

void infixToPostfix(char infix[], char postfix[]) {
    int i = 0, k = 0;
    char ch, elem;

    push('#');

    while ((ch = infix[i++]) != '\0') {
        if (ch == '(') {
            push(ch);
        } else if (isalnum(ch)) {
            postfix[k++] = ch;
        } else if (ch == ')') {
            while (stack[top] != '(') {
                if (top == -1) {
                    printf("\nError: Unbalanced parentheses!");
                    return;
                }
                postfix[k++] = pop();
            }
            elem = pop();
        } else {
            while (pr(stack[top]) >= pr(ch)) {
                postfix[k++] = pop();
            }
            push(ch);
        }
    }

    while (stack[top] != '#') { 
        postfix[k++] = pop();
        if (top == -1) {
            printf("\nError: Unbalanced expression after processing!");
            return;
        }
    }

    postfix[k] = '\0';
}

int main() {
    char infix[SIZE], postfix[SIZE];

    printf("\n\n\nEnter Infix Expression: ");
    if (scanf("%s", infix) != 1) { 
        return 1;
    }

    infixToPostfix(infix, postfix);

    printf("\nPostfix Expression = %s\n", postfix);

    return 0;
}
