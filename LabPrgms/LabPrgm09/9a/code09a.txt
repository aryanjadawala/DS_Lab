#include <stdio.h>

int a[10][10], vis[10], n;

void bfs(int v)
{
    int q[10], f = 0, r = 0, i;
    q[r++] = v;
    vis[v] = 1;

    while (f < r)
    {
        v = q[f++];
        printf("%d ", v);

        for (i = 1; i <= n; i++)
        {
            if (a[v][i] == 1 && vis[i] == 0)
            {
                vis[i] = 1;
                q[r++] = i;
            }
        }
    }
}

int main()
{
    int i, j, src;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");
    for (i = 1; i <= n; i++)
    {
        for (j = 1; j <= n; j++)
        {
            scanf("%d", &a[i][j]);
        }
        vis[i] = 0;
    }

    printf("Enter source vertex: ");
    scanf("%d", &src);

    printf("BFS Traversal:\n");
    bfs(src);

    return 0;
}
